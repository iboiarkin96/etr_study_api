 52    def summary(self, *, owner_client_uuid: str) -> ScheduleSummaryResponse:
 53        """Compute the ``/schedule/summary`` payload for one learner.
 54
 55        Args:
 56            owner_client_uuid: Resolved owner scope.
 57
 58        Returns:
 59            :class:`ScheduleSummaryResponse` with slot counts + due-window counts.
 60        """
 61        now = datetime.now(UTC)
 62        counts = self.repository.counts_by_slot(owner_client_uuid)
 63        by_slot = ScheduleSummarySlotCounts(
 64            A=counts.get("A", 0),
 65            B=counts.get("B", 0),
 66            C=counts.get("C", 0),
 67            D=counts.get("D", 0),
 68        )
 69        total = by_slot.A + by_slot.B + by_slot.C + by_slot.D
 70        due_now = self.repository.count_in_window(owner_client_uuid, start=None, end=now)
 71        due_next_24h = self.repository.count_in_window(
 72            owner_client_uuid,
 73            start=now,
 74            end=now + timedelta(hours=24),
 75        )
 76        return ScheduleSummaryResponse(
 77            by_slot=by_slot,
 78            due_now=due_now,
 79            due_next_24h=due_next_24h,
 80            total=total,
 81            computed_at=now,
 82        )

 84    def preview(
 85        self,
 86        *,
 87        owner_client_uuid: str,
 88        inputs: SchedulePreviewInputs,
 89    ) -> SchedulePreviewResponse:
 90        """Compute the ``/schedule/preview`` payload — window scan + deterministic shuffle.
 91
 92        Determinism boundary: same ``random_seed`` within the same clock minute yields the
 93        same order. A minute later the derived seed changes so the client can refresh.
 94
 95        Args:
 96            owner_client_uuid: Resolved owner scope.
 97            inputs: Validated request parameters.
 98
 99        Returns:
100            :class:`SchedulePreviewResponse` with 0..``limit`` items and ``random_seed``
101            echoed (server-generated when omitted).
102        """
103        now = datetime.now(UTC)
104        window_delta = _WINDOW_DURATIONS[inputs.window]
105        until = now + window_delta
106        candidates = self.repository.preview_candidates(owner_client_uuid, until=until)
107        seed = inputs.random_seed if inputs.random_seed is not None else _generate_seed()
108
109        indices = list(range(len(candidates)))
110        random.Random(_derive_seed(seed, now)).shuffle(indices)
111        top = indices[: inputs.limit]
112
113        items = [
114            SchedulePreviewItem(
115                preview_order_index=order_index,
116                conspectus_uuid=candidates[picked].conspectus_uuid,
117                title=candidates[picked].title,
118                slot=candidates[picked].slot,
119                next_review_at=candidates[picked].next_review_at,
120            )
121            for order_index, picked in enumerate(top)
122        ]
123        return SchedulePreviewResponse(
124            window=inputs.window,
125            computed_at=now,
126            random_seed=seed,
127            count=len(items),
128            items=items,
129        )

132def _generate_seed() -> str:
133    """Produce a URL-safe base64 22-char slug from a random UUIDv4."""
134    return base64.urlsafe_b64encode(uuid.uuid4().bytes).decode("ascii").rstrip("=")

137def _derive_seed(seed: str, moment: datetime) -> int:
138    """Combine ``seed`` with the current clock minute to drive a deterministic shuffle.
139
140    Args:
141        seed: Client-supplied or server-generated seed slug.
142        moment: Reference timestamp (must be UTC-aware).
143
144    Returns:
145        64-bit integer seed suitable for :class:`random.Random`.
146    """
147    key = f"{seed}|{moment.strftime('%Y-%m-%dT%H:%M')}".encode("utf-8")
148    digest = hashlib.blake2b(key, digest_size=8).digest()
149    return int.from_bytes(digest, "big", signed=False)
