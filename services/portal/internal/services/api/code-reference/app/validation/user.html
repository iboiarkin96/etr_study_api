 18def _json_safe(value: Any) -> Any:
 19    """Recursively coerce values for inclusion in JSON error ``details``.
 20
 21    Args:
 22        value: Arbitrary object from Pydantic error context.
 23
 24    Returns:
 25        JSON-compatible structure; non-JSON types become ``str(value)``.
 26    """
 27    if value is None or isinstance(value, (str, int, float, bool)):
 28        return value
 29    if isinstance(value, dict):
 30        return {str(k): _json_safe(v) for k, v in value.items()}
 31    if isinstance(value, (list, tuple, set)):
 32        return [_json_safe(v) for v in value]
 33    return str(value)

 36def _field_from_loc(loc: list[Any]) -> str | None:
 37    """Extract JSON body field name from a Pydantic ``loc`` tuple when applicable.
 38
 39    Args:
 40        loc: Location path from :meth:`RequestValidationError.errors`.
 41
 42    Returns:
 43        Body field name, or ``None`` if not a simple body field error.
 44    """
 45    if len(loc) >= 2 and loc[0] == "body" and isinstance(loc[1], str):
 46        return loc[1]
 47    return None

 50def _resolve_update_user_rule(field: str | None, error_type: str) -> StableError:
 51    """Map ``(field, error_type)`` for PUT/PATCH user-by-composite-key bodies.
 52
 53    Args:
 54        field: Body field name from :func:`_field_from_loc`, or ``None``.
 55        error_type: Pydantic error type string (e.g. ``missing``, ``string_too_short``).
 56
 57    Returns:
 58        Matching rule from :data:`~app.errors.user.UPDATE_USER_VALIDATION_RULES` or
 59        :data:`~app.errors.common.COMMON_000`.
 60    """
 61    if field is not None and (field, error_type) in UPDATE_USER_VALIDATION_RULES:
 62        return UPDATE_USER_VALIDATION_RULES[(field, error_type)]
 63    return COMMON_000

 66def _resolve_create_user_rule(field: str | None, error_type: str) -> StableError:
 67    """Map ``(field, error_type)`` to a rule or fall back to a generic validation code.
 68
 69    Args:
 70        field: Body field name from :func:`_field_from_loc`, or ``None``.
 71        error_type: Pydantic error type string (e.g. ``missing``, ``string_too_short``).
 72
 73    Returns:
 74        Matching rule from :data:`~app.errors.user.CREATE_USER_VALIDATION_RULES` or
 75        :data:`~app.errors.common.COMMON_000`.
 76    """
 77    if field is not None and (field, error_type) in CREATE_USER_VALIDATION_RULES:
 78        return CREATE_USER_VALIDATION_RULES[(field, error_type)]
 79    return COMMON_000

 82def build_validation_error_payload(request: Request, exc: RequestValidationError) -> dict[str, Any]:
 83    """Normalize FastAPI/Pydantic validation errors into the API 422 contract.
 84
 85    Args:
 86        request: Failed request (method/path select specialized rules).
 87        exc: Validation exception from FastAPI.
 88
 89    Returns:
 90        Dict with ``error_type``, ``endpoint``, and ``errors`` list suitable for JSON encoding.
 91    """
 92    errors: list[dict[str, Any]] = []
 93    endpoint = f"{request.method} {request.url.path}"
 94
 95    for item in exc.errors():
 96        loc = list[Any](item.get("loc", []))
 97        error_type = str(item.get("type", "value_error"))
 98        field = _field_from_loc(loc)
 99
100        if endpoint == "POST /api/v1/user":
101            rule = _resolve_create_user_rule(field, error_type)
102        elif request.method in ("PUT", "PATCH") and PUT_PATCH_USER_BY_COMPOSITE_PATH.match(
103            request.url.path
104        ):
105            rule = _resolve_update_user_rule(field, error_type)
106        else:
107            rule = COMMON_000
108
109        errors.append(
110            {
111                "code": rule.code,
112                "key": rule.key,
113                "message": rule.message,
114                "field": field,
115                "source": "validation",
116                "details": {
117                    "type": error_type,
118                    "loc": _json_safe(loc),
119                    "input": _json_safe(item.get("input")),
120                    "ctx": _json_safe(item.get("ctx")),
121                },
122            }
123        )
124
125    return {"error_type": "validation_error", "endpoint": endpoint, "errors": errors}
