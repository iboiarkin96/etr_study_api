19def enrich_openapi_with_request_id(schema: dict[str, Any]) -> None:
20    """Mutate ``schema`` so every operation documents request/response ``X-Request-Id``.
21
22    Swagger UI will show an optional header parameter and list the response header on each
23    status, so Try it out flows match curl and cross-origin browser clients.
24
25    Args:
26        schema: OpenAPI document produced by :func:`fastapi.openapi.utils.get_openapi`.
27    """
28    response_header: dict[str, Any] = {
29        "description": _RESPONSE_HEADER_DESC,
30        "schema": {"type": "string"},
31    }
32    request_parameter: dict[str, Any] = {
33        "name": "X-Request-Id",
34        "in": "header",
35        "required": False,
36        "schema": {"type": "string"},
37        "description": _REQUEST_ID_HEADER_DESC,
38    }
39
40    paths = schema.get("paths")
41    if not isinstance(paths, dict):
42        return
43
44    for _path, path_item in paths.items():
45        if not isinstance(path_item, dict):
46            continue
47        for method, operation in path_item.items():
48            if method not in (
49                "get",
50                "post",
51                "put",
52                "patch",
53                "delete",
54                "options",
55                "head",
56                "trace",
57            ):
58                continue
59            if not isinstance(operation, dict):
60                continue
61
62            params = operation.setdefault("parameters", [])
63            if not _has_header_param(params, "X-Request-Id"):
64                params.append(request_parameter.copy())
65
66            responses = operation.get("responses")
67            if not isinstance(responses, dict):
68                continue
69            for _code, resp in responses.items():
70                if not isinstance(resp, dict):
71                    continue
72                hdrs = resp.setdefault("headers", {})
73                hdrs.setdefault("X-Request-Id", response_header.copy())

76def _has_header_param(parameters: list[Any], name: str) -> bool:
77    """Return True if ``parameters`` already contains a header named ``name``."""
78    for item in parameters:
79        if not isinstance(item, dict):
80            continue
81        if item.get("in") == "header" and item.get("name") == name:
82            return True
83    return False
