27        """Load a stored idempotency record for a route and key.
28
29        Args:
30            endpoint_path: Normalized path (e.g. ``/api/v1/user``).
31            idempotency_key: Client-supplied idempotency token.
32
33        Returns:

35        """
36        stmt = select(IdempotencyKeyRecord).where(

39        )
40        row = self.session.execute(stmt).scalar_one_or_none()
41        if row is None:
42            return None
43        return IdempotencyRecord(
44            payload_hash=row.payload_hash,

46            response_body=json.loads(row.response_body),
47        )
