13def _check_timezone(value: str) -> str:
14    """``AfterValidator`` callback: ensure ``value`` is a known IANA zone name.
15
16    Args:
17        value: Candidate timezone string from the request body.
18
19    Returns:
20        The same string if valid.
21
22    Raises:
23        ValueError: If ``value`` is not in :data:`VALID_TIMEZONES`.
24    """
25    if value not in VALID_TIMEZONES:
26        raise ValueError(
27            f"Unknown timezone '{value}'. "
28            "Use a valid IANA timezone (e.g. 'UTC', 'Europe/Moscow', 'America/New_York')."
29        )
30    return value
