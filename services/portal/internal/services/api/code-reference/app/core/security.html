 49    def __init__(self, limit: int, window_seconds: int) -> None:
 50        """Initialize limiter state.
 51
 52        Args:
 53            limit: Maximum number of allowed hits per ``bucket`` within one window.
 54            window_seconds: Rolling window length in seconds (monotonic clock).
 55        """
 56        self._limit = limit
 57        self._window_seconds = window_seconds
 58        self._hits: dict[str, deque[float]] = defaultdict(deque)
 59        self._lock = Lock()
 60
 61    def check(self, bucket: str) -> RateLimitResult:
 62        """Record one hit for ``bucket`` and return whether the limit is exceeded.
 63
 64        Args:
 65            bucket: Arbitrary key (e.g. ``client_id:path``).
 66
 67        Returns:
 68            Allow/deny decision with optional retry hint.
 69        """
 70        now = monotonic()
 71        window_start = now - self._window_seconds
 72        with self._lock:
 73            events = self._hits[bucket]
 74            while events and events[0] <= window_start:
 75                events.popleft()
 76
 77            if len(events) >= self._limit:
 78                retry_after = max(1, int(self._window_seconds - (now - events[0])))
 79                return RateLimitResult(
 80                    allowed=False,
 81                    remaining=0,
 82                    retry_after_seconds=retry_after,
 83                )
 84
 85            events.append(now)
 86            remaining = max(0, self._limit - len(events))
 87            return RateLimitResult(
 88                allowed=True,
 89                remaining=remaining,
 90                retry_after_seconds=0,
 91            )

 94def apply_security_headers(response: Response, request_path: str) -> None:
 95    """Set baseline security headers; relax CSP slightly for Swagger UI paths.
 96
 97    Args:
 98        response: Outgoing Starlette/FastAPI response (mutated in place).
 99        request_path: URL path, used to choose Content-Security-Policy for ``/docs``.
100    """
101    response.headers["X-Content-Type-Options"] = "nosniff"
102    response.headers["X-Frame-Options"] = "DENY"
103    response.headers["Referrer-Policy"] = "no-referrer"
104    response.headers["Permissions-Policy"] = "camera=(), microphone=(), geolocation=()"
105    if request_path.startswith("/docs") or request_path.startswith("/redoc"):
106        # Swagger/ReDoc load assets from jsdelivr; keep policy strict but compatible.
107        response.headers["Content-Security-Policy"] = (
108            "default-src 'self'; "
109            "base-uri 'none'; "
110            "frame-ancestors 'none'; "
111            "form-action 'none'; "
112            "script-src 'self' 'unsafe-inline' https://cdn.jsdelivr.net; "
113            "style-src 'self' 'unsafe-inline' https://cdn.jsdelivr.net; "
114            "img-src 'self' data: https://fastapi.tiangolo.com; "
115            "font-src 'self' data: https://cdn.jsdelivr.net; "
116            "connect-src 'self'"
117        )
118        return
119
120    response.headers["Content-Security-Policy"] = (
121        "default-src 'none'; frame-ancestors 'none'; base-uri 'none'; form-action 'none'"
122    )

125def extract_client_id(request: Request) -> str:
126    """Derive a stable client identifier for rate limiting and logging.
127
128    Args:
129        request: Incoming ASGI request.
130
131    Returns:
132        First address from ``X-Forwarded-For``, else ``request.client.host``, else a fallback string.
133    """
134    forwarded = request.headers.get("x-forwarded-for")
135    if forwarded:
136        return forwarded.split(",")[0].strip()
137    if request.client and request.client.host:
138        return request.client.host
139    return "unknown-client"

158def authenticate_request(request: Request, settings: Settings) -> JSONResponse | None:
159    """Validate credentials according to ``settings.api_auth_strategy``.
160
161    Args:
162        request: Incoming ASGI request (headers inspected).
163        settings: Auth strategy and secret configuration.
164
165    Returns:
166        ``None`` if the request is allowed, otherwise a ready-to-send error
167        :class:`~starlette.responses.JSONResponse` (401 or 500).
168    """
169    strategy = settings.api_auth_strategy.strip().lower()
170    if strategy in {"disabled", "none", "off"}:
171        return None
172
173    if strategy == "mock_api_key":
174        provided_key = request.headers.get(settings.api_auth_header)
175        if provided_key == settings.api_mock_api_key:
176            return None
177        return JSONResponse(
178            status_code=401,
179            content={
180                "detail": build_security_error_payload(
181                    COMMON_401,
182                    message=(f"Missing or invalid API key in header `{settings.api_auth_header}`."),
183                )
184            },
185        )
186
187    return JSONResponse(
188        status_code=500,
189        content={
190            "detail": build_security_error_payload(
191                COMMON_500,
192                message=f"Unsupported auth strategy: `{settings.api_auth_strategy}`.",
193            )
194        },
195    )
