 43    def __init__(self, db_path: str) -> None:
 44        """Initialize store and ensure schema.
 45
 46        Args:
 47            db_path: SQLite file path used by the application runtime.
 48        """
 49        path = Path(db_path).expanduser()
 50        if not path.is_absolute():
 51            path = Path.cwd() / path
 52        path.parent.mkdir(parents=True, exist_ok=True)
 53        self._db_path = path
 54        self._lock = Lock()
 55        self._ensure_schema()

 57    def _connection(self) -> sqlite3.Connection:
 58        """Return a configured SQLite connection.
 59
 60        Returns:
 61            Open connection with row access by column name.
 62        """
 63        conn = sqlite3.connect(self._db_path, timeout=5.0, check_same_thread=False)
 64        conn.row_factory = sqlite3.Row
 65        return conn

 67    @contextmanager
 68    def _managed_connection(self) -> Iterator[sqlite3.Connection]:
 69        """Yield a SQLite connection that is always closed.
 70
 71        Yields:
 72            Open SQLite connection object.
 73        """
 74        conn = self._connection()
 75        try:
 76            yield conn
 77        finally:
 78            conn.close()

 80    def _ensure_schema(self) -> None:
 81        """Create telemetry schema and indexes if absent."""

 84            conn.execute(
 85                """

 87                    id INTEGER PRIMARY KEY AUTOINCREMENT,
 88                    emitted_at_ms INTEGER NOT NULL,
 89                    event TEXT NOT NULL,
 90                    page_path TEXT,
 91                    session_id TEXT,
 92                    query_id TEXT,
 93                    query_text TEXT,
 94                    query_len INTEGER,
 95                    tokens_count INTEGER,
 96                    results_count INTEGER,
 97                    result_rank INTEGER,
 98                    result_url TEXT,
 99                    latency_ms INTEGER,
100                    time_to_success_ms INTEGER,
101                    time_to_click_ms INTEGER,
102                    source TEXT,
103                    top_results_json TEXT,
104                    payload_json TEXT NOT NULL
105                )
106                """
107            )
108            conn.execute(
109                "CREATE INDEX IF NOT EXISTS idx_docs_search_events_event_ts "
110                "ON docs_search_events(event, emitted_at_ms)"
111            )
112            conn.execute(
113                "CREATE INDEX IF NOT EXISTS idx_docs_search_events_query_id "
114                "ON docs_search_events(query_id)"
115            )
116            conn.execute(
117                "CREATE INDEX IF NOT EXISTS idx_docs_search_events_session_id "
118                "ON docs_search_events(session_id)"
119            )
120            conn.commit()

178    def metrics(self, *, now_ms: int, window_minutes: int) -> DocsSearchTelemetryMetrics:
179        """Compute search KPI aggregates for the requested horizon.
180
181        Args:
182            now_ms: Current epoch milliseconds.
183            window_minutes: Rolling aggregation window size.
184
185        Returns:
186            Aggregated docs-search KPI set.
187        """
188        since_ms = max(0, now_ms - (window_minutes * 60_000))
189        with self._lock, self._managed_connection() as conn:
190            total_queries = int(
191                conn.execute(
192                    """
193                    SELECT COUNT(*) AS c
194                    FROM docs_search_events
195                    WHERE event = 'search_query' AND emitted_at_ms >= ?
196                    """,
197                    (since_ms,),
198                ).fetchone()["c"]
199            )
200            zero_result_queries = int(
201                conn.execute(
202                    """
203                    SELECT COUNT(*) AS c
204                    FROM docs_search_events
205                    WHERE event = 'search_query' AND emitted_at_ms >= ? AND results_count = 0
206                    """,
207                    (since_ms,),
208                ).fetchone()["c"]
209            )
210            queries_with_click = int(
211                conn.execute(
212                    """
213                    SELECT COUNT(DISTINCT q.query_id) AS c
214                    FROM docs_search_events q
215                    WHERE q.event = 'search_query'
216                      AND q.emitted_at_ms >= ?
217                      AND EXISTS (
218                          SELECT 1
219                          FROM docs_search_events c
220                          WHERE c.event = 'search_result_click'
221                            AND c.query_id = q.query_id
222                            AND c.emitted_at_ms >= ?
223                      )
224                    """,
225                    (since_ms, since_ms),
226                ).fetchone()["c"]
227            )
228            success_times = [
229                int(row["time_to_success_ms"])
230                for row in conn.execute(
231                    """
232                    SELECT MIN(time_to_success_ms) AS time_to_success_ms
233                    FROM docs_search_events
234                    WHERE event = 'search_success'
235                      AND emitted_at_ms >= ?
236                      AND time_to_success_ms > 0
237                    GROUP BY session_id
238                    ORDER BY time_to_success_ms ASC
239                    """,
240                    (since_ms,),
241                ).fetchall()
242                if row["time_to_success_ms"] is not None
243            ]
244
245        def _percentile(values: list[int], p: float) -> int | None:
246            if not values:
247                return None
248            idx = max(0, min(len(values) - 1, int(round((len(values) - 1) * p))))
249            return values[idx]
250
251        return DocsSearchTelemetryMetrics(
252            window_minutes=window_minutes,
253            total_queries=total_queries,
254            zero_result_queries=zero_result_queries,
255            zero_result_rate=(zero_result_queries / total_queries) if total_queries > 0 else 0.0,
256            queries_with_click=queries_with_click,
257            query_ctr=(queries_with_click / total_queries) if total_queries > 0 else 0.0,
258            median_time_to_first_success_ms=_percentile(success_times, 0.50),
259            p75_time_to_first_success_ms=_percentile(success_times, 0.75),
260        )

246        def _percentile(values: list[int], p: float) -> int | None:
247            if not values:
248                return None
249            idx = max(0, min(len(values) - 1, int(round((len(values) - 1) * p))))
250            return values[idx]
251
252        return DocsSearchTelemetryMetrics(
253            window_minutes=window_minutes,
254            total_queries=total_queries,
255            zero_result_queries=zero_result_queries,
256            zero_result_rate=(zero_result_queries / total_queries) if total_queries > 0 else 0.0,
257            queries_with_click=queries_with_click,
258            query_ctr=(queries_with_click / total_queries) if total_queries > 0 else 0.0,
259            median_time_to_first_success_ms=_percentile(success_times, 0.50),
260            p75_time_to_first_success_ms=_percentile(success_times, 0.75),
261        )
