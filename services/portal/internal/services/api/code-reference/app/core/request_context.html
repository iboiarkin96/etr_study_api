16def normalize_request_id_header(raw: str | None) -> str | None:
17    """Return a safe request id from ``X-Request-Id`` or ``None`` if invalid or empty.
18
19    Args:
20        raw: Header value from the client.
21
22    Returns:
23        Stripped id when it matches allowed characters and length, else ``None``.
24    """
25    if not raw:
26        return None
27    candidate = raw.strip()
28    if not candidate or len(candidate) > 128:
29        return None
30    if _REQUEST_ID_RE.match(candidate):
