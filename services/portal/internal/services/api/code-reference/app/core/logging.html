 18class RequestContextFilter(logging.Filter):
 19    """Attach correlation fields from contextvars for formatters."""
 20
 21    def filter(self, record: logging.LogRecord) -> bool:
 22        rid = request_id_var.get(None)
 23        record.request_id = rid if rid else "-"
 24        tid = trace_id_var.get(None)
 25        record.trace_id = tid if tid else ""
 26        sid = span_id_var.get(None)
 27        record.span_id = sid if sid else ""
 28        return True

 31class JsonLogFormatter(logging.Formatter):
 32    """One JSON object per line (NDJSON), suitable for Filebeat → Elasticsearch."""
 33
 34    def __init__(self, *, service_name: str, app_env: str) -> None:
 35        super().__init__()
 36        self._service_name = service_name
 37        self._app_env = app_env
 38
 39    def format(self, record: logging.LogRecord) -> str:
 40        """Serialize ``record`` to a single JSON line with ECS-friendly keys.
 41
 42        Args:
 43            record: Log record after :class:`RequestContextFilter` ran.
 44
 45        Returns:
 46            UTF-8 JSON object as a string (no trailing newline; StreamHandler adds one).
 47        """
 48        ts = datetime.fromtimestamp(record.created, tz=UTC)
 49        payload: dict[str, object] = {
 50            "@timestamp": ts.strftime("%Y-%m-%dT%H:%M:%S.%f")[:-3] + "Z",
 51            "level": record.levelname,
 52            "logger": record.name,
 53            "message": record.getMessage(),
 54            "service": {"name": self._service_name},
 55            "app_env": self._app_env,
 56            "request_id": getattr(record, "request_id", "-"),
 57        }
 58        trace_id = getattr(record, "trace_id", "") or None
 59        span_id = getattr(record, "span_id", "") or None
 60        payload["trace_id"] = trace_id
 61        payload["span_id"] = span_id
 62        if record.exc_info:
 63            payload["error"] = self.formatException(record.exc_info).strip()
 64        return json.dumps(payload, ensure_ascii=False)

 67class TextLogFormatter(logging.Formatter):
 68    """Plain text with optional ``exc_info`` appended."""
 69
 70    def __init__(self) -> None:
 71        super().__init__(fmt=TEXT_LOG_FORMAT, datefmt=DATE_FORMAT)
 72
 73    def format(self, record: logging.LogRecord) -> str:
 74        line = super().format(record)
 75        if record.exc_info:
 76            line += "\n" + self.formatException(record.exc_info).strip()
 77        return line

 80def _resolve_log_path(settings: Settings) -> Path:
 81    """Resolve and ensure the directory for the rotating log file exists.
 82
 83    Args:
 84        settings: Configuration providing ``log_dir`` and ``log_file_name``.
 85
 86    Returns:
 87        Absolute path to the log file (parent directories created as needed).
 88    """
 89    log_dir = Path(settings.log_dir).expanduser()
 90    if not log_dir.is_absolute():
 91        log_dir = Path.cwd() / log_dir
 92    log_dir.mkdir(parents=True, exist_ok=True)
 93    return log_dir / settings.log_file_name

 96def configure_logging(settings: Settings) -> Path:
 97    """Reset root logging to a single timed rotating file and propagate framework logs.
 98
 99    Args:
100        settings: Log level, directory, file name, and ``log_format`` (``text`` or ``json``).
101
102    Returns:
103        Absolute path of the active log file.
104    """
105    log_path = _resolve_log_path(settings)
106    level = getattr(logging, settings.log_level, logging.INFO)
107
108    root_logger = logging.getLogger()
109    root_logger.setLevel(level)
110
111    for handler in list(root_logger.handlers):
112        root_logger.removeHandler(handler)
113
114    file_handler = TimedRotatingFileHandler(
115        filename=str(log_path),
116        when="midnight",
117        backupCount=14,
118        encoding="utf-8",
119    )
120    file_handler.setLevel(level)
121    ctx_filter = RequestContextFilter()
122    file_handler.addFilter(ctx_filter)
123    if settings.log_format == "json":
124        file_handler.setFormatter(
125            JsonLogFormatter(service_name=settings.log_service_name, app_env=settings.app_env)
126        )
127    else:
128        file_handler.setFormatter(TextLogFormatter())
129    root_logger.addHandler(file_handler)
130
131    logging.captureWarnings(True)
132
133    for logger_name in ("uvicorn", "uvicorn.error", "uvicorn.access", "fastapi"):
134        framework_logger = logging.getLogger(logger_name)
135        framework_logger.handlers = []
136        framework_logger.propagate = True
137
138    return log_path

Format the specified record as text.

The record's attribute dictionary is used as the operand to a string formatting operation which yields the returned string. Before formatting the dictionary, a couple of preparatory steps are carried out. The message attribute of the record is computed using LogRecord.getMessage(). If the formatting string uses the time (as determined by a call to usesTime(), formatTime() is called to format the event time. If there is exception information, it is formatted using formatException() and appended to the message.
